292 ]);
293 $entreprise->libelle = $validated['libelle'];

296 $entreprise->description = $validated['description'];
297 $entreprise->email = $validated['email'];
298 $entreprise->numTel = $validated['numTel'];
299
300 $typeRdv = $entreprise->typeRdv ? json_decode($entreprise->typeRdv, true) : [];
301 // Mise à jour des réponses
302 $typeRdv[0] = $request->input('question_0', $typeRdv[0] ?? 0);
303 $typeRdv[1] = $request->input('question_1', $typeRdv[1] ?? 0);

78 public function store(Request $request, Entreprise $entreprise, Activite $activite): RedirectResponse
79 {
80 // Validation des données
81 $validated = $request->validate([
82 'dateRdv' => 'required|date_format:Y-m-d',
83 'horaire' => 'required|string',
84 'nbPersonnes' => 'nullable|integer|min:1',
85 'notifications' => 'sometimes|array',
86 'notifications.*.typeNotification' => 'sometimes|string|in:SMS,Mail',
87 'notifications.*.contenu' => 'sometimes|string',
88 'notifications.*.duree' => 'sometimes|string|in:1jour,2jours,1semaine',
89 'employe_id' => 'required|integer|exists:users,id',
90 ]);
91
92 // Extraction de l'horaire
93 [$heureDeb, $heureFin] = explode(' - ', $validated['horaire']);
94
95 // Vérification et suppression du rendez-vous précédent pour cet utilisateur
96 $previousReservation = Auth::user()
97 ->effectuer_reservations()
98 ->whereHas('effectuer_activites', function ($query) use ($activite) {
99 $query->where('idActivite', $activite->id);
100 })
101 ->first();
102
103 if ($previousReservation) {
104 $previousReservation->notifications()->each(function ($notification) {
105 $notification->delete();
106 });
107
108 $previousReservation->effectuer_activites()->detach();
109 $previousReservation->delete();
110 }
111
112
113 if ($previousReservation) {
114 // Supprimer les relations et notifications liées à la réservation précédente
115 $previousReservation->notifications()->each(function ($notification) {
116 $notification->delete();
117 });
118
119 $previousReservation->effectuer_activites()->detach();
120
121 // Supprimez ensuite la réservation
122 $previousReservation->delete();
123 }
124
125
126 // Création de la nouvelle réservation
127 $reservation = Reservation::create([
128 'dateRdv' => $validated['dateRdv'],
129 'heureDeb' => Carbon::parse($heureDeb)->format('H:i:s'),
130 'heureFin' => Carbon::parse($heureFin)->format('H:i:s'),
131 'nbPersonnes' => $validated['nbPersonnes'] ?? 1,
132 ]);
133
134 // Gestion des notifications
135 foreach ($validated['notifications'] ?? [] as $notificationData) {
136 $notification = new Notification([
137 'categorie' => $notificationData['typeNotification'],
138 'contenu' => $notificationData['contenu'],
139 'delai' => match ($notificationData['duree']) {
140 '1jour' => 24,
141 '2jours' => 48,
142 '1semaine' => 168,
143 },
144 'etat' => 0,
145 'reservation_id' => $reservation->id
146 ]);
147
148 $reservation->notifications()->save($notification);
149 }
150
151 // Lier la réservation à l'activité et à l'utilisateur
152 $reservation->effectuer_activites()->attach($activite->id, [
153 'idUser' => Auth::id(),
154 'dateReservation' => now(),
155 'typeNotif' => 'SMS',
156 'numTel' => Auth::user()->numTel,
157 ]);
158
159 return redirect()
160 ->route('reservation.index')
161 ->with('success', 'Votre réservation a été enregistrée avec succès !');
162 }

206 public function update(Request $request, Reservation $reservation): RedirectResponse
207 {
208 // Valider les données
209 $validated = $request->validate([
210 'slot' => 'required|string', // Ex: "09:00 - 10:00|2025-02-01"
211 'nbPersonnes' => 'integer|min:1',
212 ]);
213
214 // 1) Extraire l'horaire et la date depuis 'slot'
215 [$timeRange, $jour] = explode('|', $validated['slot']);
216 [$hDeb, $hFin] = explode(' - ', $timeRange);
217
218 // 2) Créer la nouvelle réservation
219 $newReservation = Reservation::create([
220 'dateRdv' => $jour,
221 'heureDeb' => Carbon::parse($hDeb)->format('H:i:s'),
222 'heureFin' => Carbon::parse($hFin)->format('H:i:s'),
223 'nbPersonnes' => $reservation->nbPersonnes,
224 ]);
225
226 // 3) Récupérer l’activité liée à l’ancienne réservation
227 $activite = $reservation->effectuer_activites()->first();
228
229 // 4) Attacher la nouvelle réservation (table pivot 'effectuer')
230 $newReservation->effectuer_activites()->attach($activite->id, [
231 'idUser' => Auth::id(),
232 'dateReservation' => now(),
233 'typeNotif' => 'SMS',
234 'numTel' => Auth::user()->numTel,
235 ]);
236
237 // 5) Supprimer l’ancienne réservation et ses relations
238 $this->destroy($reservation);
239
240 return redirect()
241 ->route('reservation.index')
242 ->with('success', 'Votre réservation a été modifiée avec succès !');
243 }

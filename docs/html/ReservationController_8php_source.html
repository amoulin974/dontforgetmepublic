171 // Create the new reservation with the validated data.

173 'dateRdv' => $validated['dateRdv'],
174 'heureDeb' => Carbon::parse($heureDeb)->format('H:i:s'),
175 'heureFin' => Carbon::parse($heureFin)->format('H:i:s'),

339 public function calculateTimeSlots(
340 \Illuminate\Support\Collection $slots,
341 array $reservations,
342 object $company,
343 object $activity,
344 string $date
345 ): array {
346 $timeSlots = [];
347
348 foreach ($slots as $slot) {
349 $plageStart = Carbon::parse($slot->datePlage)->setTimeFromTimeString($slot->heureDeb);
350 $plageEnd = Carbon::parse($slot->datePlage)->setTimeFromTimeString($slot->heureFin);
351 if ($plageEnd->lessThan(now())) continue;
352
353 if (!$slot->relationLoaded('activites') || $slot->activites->isEmpty()) continue;
354
355 try {
356 $startTime = $plageStart;
357 $endTime = $plageEnd;
358 $interval = Carbon::parse($activity->duree)->hour * 60 + Carbon::parse($activity->duree)->minute;
359 } catch (\Exception $e) {
360 continue;
361 }
362
363 while ($startTime->lessThan($endTime)) {
364 $currentStart = Carbon::parse($slot->datePlage)->setTimeFromTimeString($startTime->format('H:i:s'));
365 $currentEnd = $currentStart->copy()->addMinutes($interval);
366
367 if ($currentStart->lessThan(now())) {
368 $startTime->addMinutes($interval);
369 continue;
370 }
371
372 $totalReservations = collect($reservations)->filter(function ($res) use ($currentStart, $currentEnd) {
373 $dateRdv = Carbon::parse($res['dateRdv'])->format('Y-m-d');
374 $resStart = Carbon::createFromFormat('Y-m-d H:i:s', "{$dateRdv} {$res['heureDeb']}");
375 $resEnd = Carbon::createFromFormat('Y-m-d H:i:s', "{$dateRdv} {$res['heureFin']}");
376 return $resStart->lt($currentEnd) && $resEnd->gt($currentStart);
377 })->sum('nbPersonnes');
378
379 $globalRemaining = $company->capaciteMax - $totalReservations;
380 $activityRemaining = $activity->nbrPlaces - $totalReservations;
381 $remainingPlaces = min($globalRemaining, $activityRemaining);
382
383 $timeSlots[] = [
384 'time_range' => $currentStart->format('H:i') . ' - ' . $currentEnd->format('H:i'),
385 'date' => $currentStart->format('Y-m-d'),
386 'remaining_places' => max(0, $remainingPlaces),
387 'start' => $currentStart->format('H:i'),
388 'end' => $currentEnd->format('H:i'),
389 ];
390
391 $startTime->addMinutes($interval);
392 }
393 }
394 return $timeSlots;
395 }

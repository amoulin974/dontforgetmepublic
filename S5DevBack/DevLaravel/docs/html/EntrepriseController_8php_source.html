124 public function ajax(Request $request)
125 {
126 switch ($request->type) {
127 case 'invite':
128 // For each activity, attach the user (identified by email) as an Invite.
129 foreach ($request->activites as $idActivite) {
130 User::where('email', $request->email)
131 ->first()
132 ->travailler_entreprises()
133 ->attach($request->idEntreprise, [
134 'idActivite' => $idActivite,
135 'statut' => 'Invité'
136 ]);
137 }
138 // Return the user information.
139 $event = User::where('email', $request->email)->first();
140 return response()->json($event);
141 break;
142
143 case 'upgrade':
144 // Upgrade the employee's status to Admin for the enterprise.
145 $event = User::where('id', $request->idEmploye)
146 ->first()
147 ->travailler_entreprises
148 ->where('id', $request->idEntreprise)
149 ->first()
150 ->pivot
151 ->update([
152 'statut' => 'Admin',
153 ]);
154 return response()->json($event);
155 break;
156
157 case 'downgrade':
158 // Downgrade the employee's status to Employé for the enterprise.
159 $event = User::where('id', $request->idEmploye)
160 ->first()
161 ->travailler_entreprises
162 ->where('id', $request->idEntreprise)
163 ->first()
164 ->pivot
165 ->update([
166 'statut' => 'Employé',
167 ]);
168 return response()->json($event);
169 break;
170
171 case 'delete':
172 // Retrieve all activity IDs related to the enterprise.
173 $activitesEntreprise = Entreprise::where('id', $request->idEntreprise)
174 ->first()
175 ->activites()
176 ->get('id');
177
178 // Detach the user from each activity.
179 foreach ($activitesEntreprise as $idActivite) {
180 $event = User::where('id', $request->idEmploye)
181 ->first()
182 ->travailler_activites()
183 ->wherePivot('idEntreprise', $request->idEntreprise)
184 ->detach($idActivite->id);
185 }
186 return response()->json($event);
187 break;
188
189 default:
190 // No action for other types.
191 break;
192 }

319 $capacity = session('capacity');
320
321 // Verify that both company and appointment data exist.

323 throw \Illuminate\Validation\ValidationException::withMessages([

326 }

329 $newCompany = Entreprise::create([

333 'adresse' => $company['rue'] . ', ' . $company['codePostal'] . ' ' . $company['ville'],
334 'description' => $company['description'] ?? 'Aucune description renseignée pour le moment.',

372 public function update(Request $request, Entreprise $entreprise)
373 {
374 // Validate the input fields.
375 $validated = $request->validate([
376 'libelle' => ['required', 'string', 'max:255'],
377 'siren' => ['required', 'string', 'max:14', 'regex:/^(\d{9}|\d{3} \d{3} \d{3})$/'],
378 'metier' => ['required', 'string', 'in:Restaurant,Coiffeur,Avocat'],
379 'rue' => ['required', 'string', 'max:255'],
380 'codePostal' => ['required', 'string', 'max:6', 'regex:/^\d{5}|\d{2} \d{3}$/'],
381 'ville' => ['required', 'string', 'max:255'],
382 'description' => ['nullable', 'string', 'max:255'],
383 'email' => ['required', 'email', 'max:255'],
384 'numTel' => ['required', 'string', 'max:15', 'regex:/^(\d{2} \d{2} \d{2} \d{2} \d{2}|\d{10})$/'],
385 'capaciteMax' => ['nullable', 'integer', 'min:1'],
386 ]);
387
388 // Update the enterprise properties.
389 $entreprise->libelle = $validated['libelle'];
390 $entreprise->siren = $validated['siren'];
391 $entreprise->metier = $validated['metier'];
392 $entreprise->adresse = $validated['rue'] . ', ' . $validated['codePostal'] . ' ' . $validated['ville'];
393 $entreprise->description = $validated['description'];
394 $entreprise->email = $validated['email'];
395 $entreprise->numTel = $validated['numTel'];
396
397 // Retrieve and update appointment type responses.
398 $typeRdv = $entreprise->typeRdv ? json_decode($entreprise->typeRdv, true) : [];
399 $typeRdv[0] = $request->input('question_0', $typeRdv[0] ?? 0);
400 $typeRdv[1] = $request->input('question_1', $typeRdv[1] ?? 0);
401 $typeRdv[2] = $request->input('question_2', $typeRdv[2] ?? 0);
402 $typeRdv[3] = $request->input('question_3', $typeRdv[3] ?? 0);
403
404 $entreprise->typeRdv = json_encode($typeRdv);
405 $entreprise->capaciteMax = ($typeRdv[0] == 0) ? 1 : $validated['capaciteMax'];
406
407 // Save the updated enterprise.
408 $entreprise->save();
409
410 return redirect()->route('entreprise.show', $entreprise)
411 ->with('success', 'Entreprise mise à jour avec succès.');
412 }

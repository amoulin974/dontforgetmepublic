98 public function store(Request $request, Entreprise $entreprise)
99 {
100 // Check if the user is an Admin.
101 $isAdmin = Auth::user()->travailler_entreprises()
102 ->wherePivot('statut', 'Admin')
103 ->wherePivot('idEntreprise', $entreprise->id)
104 ->count() > 0;
105
106 // Check if the user is the creator of the company.
107 $isCreator = $entreprise->idCreateur == Auth::user()->id;
108
109 $isAllow = $isAdmin || $isCreator;
110
111 if (!$isAllow) {
112 return redirect()->route('entreprise.index');
113 }
114 else{
115 $request->validate([
116 'libelle' => 'required|string|max:255',
117 'duree' => 'required|date_format:H:i', // Validation pour le format time
118 'nbrPlaces' => 'required|integer|min:1'
119 ]);
120
121 // Conversion de la durée du format time (HH:MM) en minutes
122 list($heures, $minutes) = explode(':', $request->duree);
123 $dureeEnMinutes = ($heures * 60) + $minutes;
124
125 // Convert duration from minutes to time format (H:i:s).
126 $dureeInTimeFormat = gmdate('H:i:s', $dureeEnMinutes * 60);
127
128 $activite = Activite::create([
129 'libelle' => $request->libelle,
130 'duree' => $dureeInTimeFormat,
131 'nbrPlaces' => $request->nbrPlaces,
132 'idEntreprise' => $entreprise->id
133 ]);
134
135 $activite->travailler_users()->attach(auth()->id(), [
136 'idEntreprise' => $entreprise->id,
137 'statut' => 'Admin',
138 ]);
139
140 // Ajouter les employés de l'entreprise à l'activité
141 $entreprise->travailler_users()->where('statut', 'Employé')->get()->each(function ($user) use ($activite, $entreprise) {
142 $activite->travailler_users()->syncWithoutDetaching([$user->id => [
143 'idEntreprise' => $entreprise->id,
144 'statut' => 'Employé',
145 ]]);
146 });
147 $entreprise->travailler_users()->where('statut', 'Admin')->get()->each(function ($user) use ($activite, $entreprise) {
148 if ($user->id != Auth::user()->id) {
149 $activite->travailler_users()->attach($user->id, [
150 'idEntreprise' => $entreprise->id,
151 'statut' => 'Admin',
152 ]);
153 }
154 });
155
156 if ($entreprise->activites()->count() === 1) {
157 $entreprise->update(['publier' => 1]);
158 }
159
160 return redirect()->route('entreprise.services.index', ['entreprise' => $entreprise->id])->with('success', 'Service créé avec succès.');
161 }

126 public function store(Request $request, Entreprise $entreprise, Activite $activite): RedirectResponse
127 {
128 // Validate the incoming request data.
129 $validated = $request->validate([
130 'dateRdv' => 'required|date_format:Y-m-d',
131 'horaire' => 'required|string',
132 'nbPersonnes' => 'nullable|integer|min:1',
133 'notifications' => 'sometimes|array',
134 'notifications.*.typeNotification' => 'sometimes|string|in:SMS,Mail',
135 'notifications.*.contenu' => 'sometimes|string',
136 'notifications.*.duree' => 'sometimes|string|in:1jour,2jours,1semaine',
137 'employe_id' => 'required|integer|exists:users,id',
138 ]);
139
140 // Extract start and end times from the "horaire" field (expected format: "HH:MM - HH:MM").
141 [$heureDeb, $heureFin] = explode(' - ', $validated['horaire']);
142
143 // Check if there is an existing reservation for this activity by the user.
144 $previousReservation = Auth::user()
145 ->effectuer_reservations()
146 ->whereHas('effectuer_activites', function ($query) use ($activite) {
147 $query->where('idActivite', $activite->id);
148 })
149 ->first();
150
151 if ($previousReservation) {
152 // Delete all notifications associated with the previous reservation.
153 $previousReservation->notifications()->each(function ($notification) {
154 $notification->delete();
155 });
156
157 // Detach pivot relationships for the previous reservation.
158 $previousReservation->effectuer_activites()->detach();
159
160 // Delete the previous reservation.
161 $previousReservation->delete();
162 }
163
164 // (Note: The deletion block appears twice in the original code.
165 // Only one block is necessary, so the duplicate code is omitted.)
166
167 // Create the new reservation with the validated data.
168 $reservation = Reservation::create([
169 'dateRdv' => $validated['dateRdv'],
170 'heureDeb' => Carbon::parse($heureDeb)->format('H:i:s'),
171 'heureFin' => Carbon::parse($heureFin)->format('H:i:s'),
172 'nbPersonnes' => $validated['nbPersonnes'] ?? 1,
173 ]);
174
175 // Process and attach notifications, if provided.
176 foreach ($validated['notifications'] ?? [] as $notificationData) {
177 $notification = new Notification([
178 'categorie' => $notificationData['typeNotification'],
179 'contenu' => $notificationData['contenu'],
180 'delai' => match ($notificationData['duree']) {
181 '1jour' => 24,
182 '2jours' => 48,
183 '1semaine'=> 168,
184 },
185 'etat' => 0,
186 'reservation_id' => $reservation->id,
187 ]);
188
189 $reservation->notifications()->save($notification);
190 }
191
192 // Attach the reservation to the activity and associate it with the authenticated user.
193 $reservation->effectuer_activites()->attach($activite->id, [
194 'idUser' => Auth::id(),
195 'dateReservation' => now(),
196 'typeNotif' => 'SMS',
197 'numTel' => Auth::user()->numTel,
198 ]);
199
200 return redirect()
201 ->route('reservation.index')
202 ->with('success', 'Votre réservation a été enregistrée avec succès !');
203 }

252 public function update(Request $request, Reservation $reservation): RedirectResponse
253 {
254 // Validate the incoming request data.

262 [$hDeb, $hFin] = explode(' - ', $timeRange);
263
264 // Create a new reservation with the updated time and date.

266 'dateRdv' => $jour,

269 'nbPersonnes' => $reservation->nbPersonnes,
270 ]);
271
272 // Retrieve the activity linked to the old reservation.
273 $activite = $reservation->effectuer_activites()->first();
274
275 // Attach the new reservation to the activity and the authenticated user.
276 $newReservation->effectuer_activites()->attach($activite->id, [
277 'idUser' => Auth::id(),
278 'dateReservation' => now(),
279 'typeNotif' => 'SMS',
280 'numTel' => Auth::user()->numTel,
281 ]);
282
283 // Delete the old reservation and its related data.
284 $this->destroy($reservation);
285
286 return redirect()

335 public function calculateTimeSlots(
336 \Illuminate\Support\Collection $slots,
337 array $reservations,
338 object $company,
339 object $activity,
340 string $date
341 ): array {
342 $timeSlots = [];
343
344 foreach ($slots as $slot) {
345 $plageStart = Carbon::parse($slot->datePlage)->setTimeFromTimeString($slot->heureDeb);
346 $plageEnd = Carbon::parse($slot->datePlage)->setTimeFromTimeString($slot->heureFin);
347 if ($plageEnd->lessThan(now())) continue;
348
349 if (!$slot->relationLoaded('activites') || $slot->activites->isEmpty()) continue;
350
351 try {
352 $startTime = $plageStart;
353 $endTime = $plageEnd;
354 $interval = Carbon::parse($activity->duree)->hour * 60 + Carbon::parse($activity->duree)->minute;
355 } catch (\Exception $e) {
356 continue;
357 }
358
359 while ($startTime->lessThan($endTime)) {
360 $currentStart = Carbon::parse($slot->datePlage)->setTimeFromTimeString($startTime->format('H:i:s'));
361 $currentEnd = $currentStart->copy()->addMinutes($interval);
362
363 if ($currentStart->lessThan(now())) {
364 $startTime->addMinutes($interval);
365 continue;
366 }
367
368 $totalReservations = collect($reservations)->filter(function ($res) use ($currentStart, $currentEnd) {
369 $dateRdv = Carbon::parse($res['dateRdv'])->format('Y-m-d');
370 $resStart = Carbon::createFromFormat('Y-m-d H:i:s', "{$dateRdv} {$res['heureDeb']}");
371 $resEnd = Carbon::createFromFormat('Y-m-d H:i:s', "{$dateRdv} {$res['heureFin']}");
372 return $resStart->lt($currentEnd) && $resEnd->gt($currentStart);
373 })->sum('nbPersonnes');
374
375 $globalRemaining = $company->capaciteMax - $totalReservations;
376 $activityRemaining = $activity->nbrPlaces - $totalReservations;
377 $remainingPlaces = min($globalRemaining, $activityRemaining);
378
379 $timeSlots[] = [
380 'time_range' => $currentStart->format('H:i') . ' - ' . $currentEnd->format('H:i'),
381 'date' => $currentStart->format('Y-m-d'),
382 'remaining_places' => max(0, $remainingPlaces),
383 'start' => $currentStart->format('H:i'),
384 'end' => $currentEnd->format('H:i'),
385 ];
386
387 $startTime->addMinutes($interval);
388 }
389 }
390 return $timeSlots;
391 }

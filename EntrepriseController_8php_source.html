139 return response()->json($event);

141

143 $event = User::where('id', $request->idEmploye)
144 ->first()
145 ->travailler_entreprises
146 ->where('id', $request->idEntreprise)
147 ->first()
148 ->pivot
149 ->update([
150 'statut' => 'Admin',
151 ]);
152 return response()->json($event);
153 break;
154
155 case 'downgrade':
156 $event = User::where('id', $request->idEmploye)
157 ->first()

159 ->where('id', $request->idEntreprise)
160 ->first()
161 ->pivot
162 ->update([
163 'statut' => 'Employé',
164 ]);
165 return response()->json($event);
166 break;
167
168 case 'delete':
169 $activitesEntreprise = Entreprise::where('id', $request->idEntreprise)
170 ->first()
171 ->activites()
172 ->get('id');
173
174 foreach ($activitesEntreprise as $idActivite) {
175 $event = User::where('id', $request->idEmploye)
176 ->first()
177 ->travailler_activites()
178 ->wherePivot('idEntreprise', $request->idEntreprise)
179 ->detach($idActivite->id);
180 }
181 return response()->json($event);
182 break;
183
184 default:
185 break;
